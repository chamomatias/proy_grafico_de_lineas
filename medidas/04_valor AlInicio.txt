VAR _FechaInicio =
    /* Obtiene el primer mes visible con datos en [01_valor].
       ALLSELECTED mantiene los filtros externos, pero elimina otros del calendario.
       FILTER descarta los meses donde la medida base está en BLANK(). */
    CALCULATE (
        MIN ( '01_dim_calendario'[mes_inicia] ),
        FILTER (
            ALLSELECTED ( '01_dim_calendario' ),
            NOT ISBLANK ( [01_valor] )          // medida de referencia
        )
    )
RETURN
/* Devuelve [01_valor] solo cuando el mes visible es el mes inicial detectado;
   para los demás meses retorna BLANK(). */
IF (
    SELECTEDVALUE ( '01_dim_calendario'[mes_inicia] ) = _FechaInicio,
    [01_valor],                                 // medida de referencia
    BLANK ()
)
